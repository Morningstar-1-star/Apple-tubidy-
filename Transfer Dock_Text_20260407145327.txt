          </div>
        </div>
      </div>
    </div>

    <!-- ADD VIDEO MODAL -->
    <div id="add-modal" class="fixed inset-0 bg-black/40 dark:bg-black/60 backdrop-blur-sm z-[100] hidden flex items-center justify-center px-4 opacity-0 transition-opacity">
      <div class="bg-white/90 dark:bg-[#1C1C1E] w-full max-w-sm rounded-[28px] p-6 shadow-2xl border border-black/10 dark:border-white/10 transform scale-95 transition-transform" id="add-modal-content">
        <h3 class="text-xl font-bold mb-4 text-black dark:text-white">Add Video</h3>
        <input type="text" id="add-url" placeholder="Video URL (.m3u8, .mp4)" class="w-full bg-gray-100 dark:bg-black/50 border border-gray-200 dark:border-white/10 rounded-xl p-4 mb-3 outline-none focus:border-accent text-black dark:text-white placeholder-gray-500">
        <input type="text" id="add-title" placeholder="Title (Optional)" class="w-full bg-gray-100 dark:bg-black/50 border border-gray-200 dark:border-white/10 rounded-xl p-4 mb-6 outline-none focus:border-accent text-black dark:text-white placeholder-gray-500">
        <div class="flex gap-3">
          <button onclick="closeAddModal()" class="flex-1 py-3.5 rounded-xl bg-gray-200 dark:bg-white/10 text-gray-800 dark:text-white font-semibold active:scale-95 transition">Cancel</button>
          <button onclick="saveVideo()" class="flex-1 py-3.5 rounded-xl bg-accent text-white font-bold active:scale-95 transition">Save</button>
        </div>
      </div>
    </div>

  </div>

  <script>
    // --- APP STATE & THEME ENGINE ---
    let videos = JSON.parse(localStorage.getItem('apple_yt_videos')) || [];
    let appSettings = JSON.parse(localStorage.getItem('apple_yt_settings')) || { theme: 'dark', accent: '#3b82f6' };
    let currentHls = null;
    let controlsTimeout = null;
    let lastTapTime = 0;
    let subtitleUrl = null;

    function initTheme() {
      setTheme(appSettings.theme, true);
      setAccent(appSettings.accent, true);
    }

    function setTheme(mode, initial = false) {
      appSettings.theme = mode;
      localStorage.setItem('apple_yt_settings', JSON.stringify(appSettings));
      
      const isDark = mode === 'dark' || (mode === 'system' && window.matchMedia('(prefers-color-scheme: dark)').matches);
      if(isDark) document.documentElement.classList.add('dark');
      else document.documentElement.classList.remove('dark');

      // Update UI Toggles
      document.querySelectorAll('#btn-theme-light, #btn-theme-dark, #btn-theme-system').forEach(btn => {
        btn.classList.remove('bg-white', 'dark:bg-[#3A3A3C]', 'shadow', 'text-black', 'dark:text-white');
        btn.classList.add('text-gray-600');
      });
      const activeBtn = document.getElementById(`btn-theme-${mode}`);
      if(activeBtn) {
        activeBtn.classList.remove('text-gray-600');
        activeBtn.classList.add('bg-white', 'dark:bg-[#3A3A3C]', 'shadow', 'text-black', 'dark:text-white');
      }
    }

    function setAccent(color, initial = false) {
      appSettings.accent = color;
      localStorage.setItem('apple_yt_settings', JSON.stringify(appSettings));
      document.documentElement.style.setProperty('--accent', color);
    }

    window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', e => {
      if(appSettings.theme === 'system') setTheme('system');
    });

    // --- DOM ELEMENTS ---
    const dom = {
      homeView: document.getElementById('home-view'),
      videoView: document.getElementById('video-view'),
      videoList: document.getElementById('video-list'),
      addModal: document.getElementById('add-modal'),
      modalContent: document.getElementById('add-modal-content'),
      appSettingsModal: document.getElementById('app-settings-modal'),
      appSettingsContent: document.getElementById('app-settings-content'),
      urlInput: document.getElementById('add-url'),
      titleInput: document.getElementById('add-title'),
      
      video: document.getElementById('main-video'),
      loader: document.getElementById('video-loader'),
      controls: document.getElementById('controls-overlay'),
      seekInput: document.getElementById('seek-input'),
      seekPlayed: document.getElementById('seek-played'),
      seekBuffered: document.getElementById('seek-buffered'),
      timeCur: document.getElementById('time-current'),
      timeTot: document.getElementById('time-total'),
      bufInfo: document.getElementById('buffer-info'),
      iconPlay: document.getElementById('icon-play'),
      iconPause: document.getElementById('icon-pause'),
      
      settings: document.getElementById('settings-panel'),
      qualSelect: document.getElementById('quality-select'),
      subUpload: document.getElementById('subtitle-upload'),
      subToggle: document.getElementById('toggle-sub-btn'),
      vidTitle: document.getElementById('vid-title')
    };

    function init() {
      initTheme();
      if (videos.length === 0) {
        videos.push({ id: Date.now().toString(), title: "Apple Event - Concept", url: "https://commondatastorage.googleapis.com/gtv-videos-bucket/sample/BigBuckBunny.mp4" });
        localStorage.setItem('apple_yt_videos', JSON.stringify(videos));
      }
      renderLibrary();
      setupPlayerEvents();
    }

    // --- APP MODALS ---
    function openAppSettings() {
      dom.appSettingsModal.classList.remove('hidden');
      setTimeout(() => { 
        dom.appSettingsModal.classList.remove('opacity-0'); 
        dom.appSettingsContent.classList.remove('translate-y-full'); 
      }, 10);
    }
    
    function closeAppSettings() {
      dom.appSettingsModal.classList.add('opacity-0'); 
      dom.appSettingsContent.classList.add('translate-y-full');
      setTimeout(() => dom.appSettingsModal.classList.add('hidden'), 300);
    }

    function openAddModal() {
      dom.addModal.classList.remove('hidden');
      setTimeout(() => { dom.addModal.classList.remove('opacity-0'); dom.modalContent.classList.remove('scale-95'); }, 10);
    }
    
    function closeAddModal() {
      dom.addModal.classList.add('opacity-0'); dom.modalContent.classList.add('scale-95');
      setTimeout(() => dom.addModal.classList.add('hidden'), 300);
      dom.urlInput.value = ''; dom.titleInput.value = '';
    }

    // --- LIBRARY ---
    function renderLibrary() {
      dom.videoList.innerHTML = '';
      if(videos.length === 0) {
        dom.videoList.innerHTML = '<p class="text-center text-gray-500 mt-10">No videos added yet.</p>'; return;
      }
      
      videos.forEach(v => {
        const div = document.createElement('div');
        div.className = "relative w-full aspect-[16/9] rounded-[28px] overflow-hidden group cursor-pointer border border-black/5 dark:border-white/5 shadow-lg bg-gray-200 dark:bg-[#151515] transition-colors";
        div.onclick = () => playVideo(v.id);
        
        div.innerHTML = `
          <div class="absolute inset-0 bg-gradient-to-br from-black/10 to-black/40 dark:from-gray-800/40 dark:to-black/80 flex items-center justify-center">
            <svg class="w-12 h-12 text-white/40 dark:text-white/20" fill="currentColor" viewBox="0 0 24 24"><path d="M8 5v14l11-7z"/></svg>
          </div>
          <div class="absolute inset-0 bg-gradient-to-t from-black/80 via-black/20 to-transparent"></div>
          
          <div class="absolute bottom-5 left-5 right-14 z-10">
            <h3 class="font-bold text-lg text-white leading-tight line-clamp-1 mb-1">${v.title || 'Untitled Video'}</h3>
            <p class="text-sm text-gray-300 dark:text-gray-400 truncate">${v.url.split('?')[0].split('/').pop()}</p>
          </div>

          <button onclick="event.stopPropagation(); deleteVideo('${v.id}')" class="absolute bottom-5 right-4 p-2.5 bg-white/20 hover:bg-red-500/80 rounded-full text-white backdrop-blur-md active:scale-90 transition-all">
             <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path></svg>
          </button>
        `;
        dom.videoList.appendChild(div);
      });
    }

    function saveVideo() {
      const url = dom.urlInput.value.trim();
      const title = dom.titleInput.value.trim() || 'Untitled Video';
      if(!url) return alert('URL is required');
      videos.unshift({ id: Date.now().toString(), title, url });
      localStorage.setItem('apple_yt_videos', JSON.stringify(videos));
      renderLibrary(); closeAddModal();
    }

    function deleteVideo(id) {
      if(confirm('Delete this video?')) {
        videos = videos.filter(v => v.id !== id);
        localStorage.setItem('apple_yt_videos', JSON.stringify(videos));
        renderLibrary();
      }
    }

    // --- PLAYER LOGIC ---
    function playVideo(id) {
      const vid = videos.find(v => v.id === id); if(!vid) return;
      dom.vidTitle.textContent = vid.title;
      dom.homeView.classList.add('view-hidden');
      dom.videoView.classList.remove('view-hidden');
      dom.videoView.classList.add('view-active');
      loadSource(vid.url);
    }

    function closeVideo() {
      dom.videoView.classList.remove('view-active');
      dom.videoView.classList.add('view-hidden');
      dom.homeView.classList.remove('view-hidden');
      setTimeout(() => {
        if(currentHls) { currentHls.destroy(); currentHls = null; }
        dom.video.pause(); dom.video.src = ''; dom.video.removeAttribute('src');
        resetPlayerUI();
      }, 400);
    }

    function loadSource(url) {
      dom.loader.classList.remove('hidden');
      if(currentHls) { currentHls.destroy(); currentHls = null; }
      resetPlayerUI();
      Array.from(dom.video.querySelectorAll('track')).forEach(t => t.remove());

      if (url.includes('.m3u8') && Hls.isSupported()) {
        currentHls = new Hls({ maxMaxBufferLength: 30 });
        currentHls.loadSource(url); currentHls.attachMedia(dom.video);
        currentHls.on(Hls.Events.MANIFEST_PARSED, (e, data) => { populateQuality(data.levels); dom.video.play().catch(e=>console.log(e)); });
        currentHls.on(Hls.Events.ERROR, (e, data) => {
          if (data.fatal) {
            if(data.type === Hls.ErrorTypes.NETWORK_ERROR) currentHls.startLoad();
            else if(data.type === Hls.ErrorTypes.MEDIA_ERROR) currentHls.recoverMediaError();
            else fallbackNative(url);
          }
        });
      } else { fallbackNative(url); }
    }

    function fallbackNative(url) {
      if(currentHls) { currentHls.destroy(); currentHls = null; }
      dom.video.src = url; dom.video.load(); dom.video.play().catch(e=>console.log(e));
      dom.qualSelect.innerHTML = '<option value="-1" class="bg-[#1C1C1E]">Native (Auto)</option>';
    }

    function resetPlayerUI() {
      dom.qualSelect.innerHTML = '<option value="-1" class="bg-[#1C1C1E]">Auto</option>';
      dom.subToggle.classList.add('hidden'); dom.subUpload.value = '';
      if(subtitleUrl) { URL.revokeObjectURL(subtitleUrl); subtitleUrl = null; }
      changeSpeed(1);
    }

    function setupPlayerEvents() {
      dom.video.addEventListener('timeupdate', updateProgress);
      dom.video.addEventListener('progress', updateBuffer);
      dom.video.addEventListener('play', () => { dom.iconPlay.classList.add('hidden'); dom.iconPause.classList.remove('hidden'); dom.loader.classList.add('hidden'); });
      dom.video.addEventListener('pause', () => { dom.iconPause.classList.add('hidden'); dom.iconPlay.classList.remove('hidden'); });
      dom.video.addEventListener('waiting', () => dom.loader.classList.remove('hidden'));
      dom.video.addEventListener('playing', () => dom.loader.classList.add('hidden'));
      dom.seekInput.addEventListener('input', (e) => {
        const t = (e.target.value / 100) * dom.video.duration;
        if(!isNaN(t)) dom.video.currentTime = t;
      });
      dom.subUpload.addEventListener('change', handleSubtitleUpload);
    }

    function togglePlay() { if(dom.video.paused) dom.video.play(); else dom.video.pause(); }

    function updateProgress() {
      if(!dom.video.duration) return;
      const pct = (dom.video.currentTime / dom.video.duration) * 100;
      dom.seekInput.value = pct; dom.seekPlayed.style.width = pct + '%';
      dom.timeCur.textContent = formatTime(dom.video.currentTime);
      dom.timeTot.textContent = formatTime(dom.video.duration);
      updateBuffer();
    }

    function updateBuffer() {
      const b = dom.video.buffered;
      if (b.length > 0 && dom.video.duration) {
        let maxEnd = 0;
        for(let i=0; i<b.length; i++) if(b.start(i) <= dom.video.currentTime && b.end(i) > dom.video.currentTime) { maxEnd = b.end(i); break; }
        dom.seekBuffered.style.width = (maxEnd / dom.video.duration) * 100 + '%';
        dom.bufInfo.textContent = `Buffered: ${Math.round(Math.max(0, maxEnd - dom.video.currentTime))}s`;
      }
    }

    function handleTap(region) {
      const now = new Date().getTime(), tapLen = now - lastTapTime;
      if(tapLen < 300 && tapLen > 0) {
        clearTimeout(controlsTimeout);
        if(region === 'left') { dom.video.currentTime = Math.max(0, dom.video.currentTime - 10); showTapIndicator('left'); }
        else if(region === 'right') { dom.video.currentTime = Math.min(dom.video.duration, dom.video.currentTime + 10); showTapIndicator('right'); }
      } else {
        controlsTimeout = setTimeout(() => {
          dom.controls.classList.toggle('active');
          if(dom.controls.classList.contains('active')) hideControlsAuto();
        }, 300);
      }
      lastTapTime = now;
    }

    function showTapIndicator(dir) {
      const ind = document.getElementById(`tap-ind-${dir}`); ind.classList.add('active');
      setTimeout(() => ind.classList.remove('active'), 500);
    }

    function hideControlsAuto() {
      clearTimeout(window.hideTimer);
      window.hideTimer = setTimeout(() => { if(!dom.video.paused) dom.controls.classList.remove('active'); }, 3000);
    }

    function toggleFullscreen() {
      const c = document.getElementById('player-container');
      if (!document.fullscreenElement && !document.webkitFullscreenElement) {
        if (c.requestFullscreen) c.requestFullscreen(); else if (c.webkitRequestFullscreen) c.webkitRequestFullscreen();
      } else {
        if (document.exitFullscreen) document.exitFullscreen(); else if (document.webkitExitFullscreen) document.webkitExitFullscreen();
      }
    }

    function toggleSettings() { dom.settings.classList.toggle('hidden'); dom.settings.classList.toggle('flex'); }

    function populateQuality(levels) {
      dom.qualSelect.innerHTML = '<option value="-1" class="bg-[#1C1C1E]">Auto</option>';
      levels.forEach((l, i) => {
        const opt = document.createElement('option'); opt.value = i; opt.className = 'bg-[#1C1C1E]';
        opt.textContent = `${l.height}p ${Math.round(l.bitrate/1000)}kbps`; dom.qualSelect.appendChild(opt);
      });
    }

    function changeQuality(val) { if(currentHls) currentHls.currentLevel = parseInt(val); toggleSettings(); }

    function changeSpeed(s) {
      dom.video.playbackRate = s;
      document.querySelectorAll('.speed-btn').forEach(btn => {
        btn.className = 'speed-btn px-4 py-2 bg-white/10 rounded-full text-sm border border-transparent text-white';
        if(btn.textContent.includes(s)) btn.className = 'speed-btn px-4 py-2 bg-accent rounded-full text-white text-sm font-bold';
      });
    }

    async function handleSubtitleUpload(e) {
      const file = e.target.files[0]; if(!file) return;
      const text = await file.text();
      let vttData = text;
      if(file.name.endsWith('.srt')) vttData = "WEBVTT\n\n" + text.replace(/\r/g, '').replace(/(\d{2}:\d{2}:\d{2}),(\d{3})/g, '$1.$2').replace(/(\d{2}:\d{2}:\d{2}),(\d{2})/g, '$1.$20').replace(/(\d{2}:\d{2}:\d{2}),(\d{1})/g, '$1.$200');
      if(subtitleUrl) URL.revokeObjectURL(subtitleUrl);
      subtitleUrl = URL.createObjectURL(new Blob([vttData], {type: 'text/vtt'}));
      Array.from(dom.video.querySelectorAll('track')).forEach(t => t.remove());
      const track = document.createElement('track');
      track.src = subtitleUrl; track.kind = 'subtitles'; track.srclang = 'en'; track.label = 'Custom'; track.default = true;
      dom.video.appendChild(track);
      dom.video.textTracks[0].mode = 'showing'; dom.subToggle.classList.remove('hidden');
    }

    function toggleSubtitles() {
      if(!dom.video.textTracks[0]) return;
      const t = dom.video.textTracks[0]; t.mode = (t.mode === 'showing') ? 'hidden' : 'showing';
    }

    function formatTime(s) {
      if(!s || isNaN(s)) return "0:00";
      const m = Math.floor(s / 60), secs = Math.floor(s % 60);
      return `${m}:${secs < 10 ? '0' : ''}${secs}`;
    }

    init();
  </script>
